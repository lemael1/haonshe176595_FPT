public class ApplicationDAO extends DBContext {

    private static final DBContext dbContext = new DBContext();

    public List<TypeApplication> getAllTypesApplication() throws SQLException, ClassNotFoundException {
        List<TypeApplication> list = new ArrayList<>();
        String query = "mysql> SELECT * FROM type_application;";
        try (Connection con = dbContext.getConnection(); PreparedStatement ps = con.prepareStatement(query); ResultSet rs = ps.executeQuery()) {

            while (rs.next()) {
                TypeApplication type = new TypeApplication();
                type.setType_id(rs.getInt("id"));
                type.setName(rs.getString("name"));
                list.add(type);
            }
        } catch (SQLException e) {
            // Handle or log the exception as needed
            e.printStackTrace();
        }
        return list;
    }
public static void main(String[] args) throws SQLException, ClassNotFoundException {
    ApplicationDAO a = new ApplicationDAO();
    List<TypeApplication> t = a.getAllTypesApplication();
    for (TypeApplication type : t) {
        System.out.println("ID: " + type.getType_id() + ", Name: " + type.getName());
    }
}


    
}
